let sparkChart=null;
function renderStockDetail(s){
  document.getElementById('stock-detail').innerHTML=`
  <div class="stock-info">
    <div style="display:flex;align-items:baseline;gap:8px">
      <span class="stock-sym">${s.sym}</span>
      <span class="stock-name">${s.name}</span>
      </div>
      <div class="stock-price ${s.chg>=0?'bull':'bear'}">$${s.price.toFixed(2)}</div>
      <div class="stock-meta">
        <span>CHG: <b class="${s.chg>=0?'bull':'bear'}">${s.chg>=0?'+':''}${s.chg.toFixed(2)} (${s.chg>=0?'+':''}${s.chgP.toFixed(2)}%)</b></span>
        <span>VOL: ${s.vol}</span>
        <span>P/E: ${s.pe}</span>
      </div>
    </div>`;
    renderSparkChart(s);
    renderAIPanel(s);
  }
  
  function renderSparkChart(s){
  const ctx=document.getElementById('sparkChart');
  if(sparkChart){sparkChart.destroy();sparkChart=null;}
  const base=s.price;
  const pts=[];
  let v=base*(1-(Math.random()*.03+.01));
  for(let i=0;i<78;i++){v+=v*(Math.random()-.48)*.008;pts.push(+v.toFixed(2));}
  pts.push(base);
  const labels=pts.map((_,i)=>{const h=Math.floor(9+i*7/78);const m=Math.floor((i*7*60/78)%60);return`${h}:${m.toString().padStart(2,'0')}`});
  const col=s.chg>=0?'#00ff9c':'#ff4d4f';
  sparkChart=new Chart(ctx,{
    type:'line',
    data:{labels,datasets:[{data:pts,borderColor:col,borderWidth:1.5,pointRadius:0,fill:true,backgroundColor:col+'18',tension:.3}]},
    options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false},tooltip:{mode:'index',intersect:false,callbacks:{label:c=>`$${c.raw.toFixed(2)}`}}},scales:{x:{display:false},y:{display:true,grid:{color:'#1e2d3d'},ticks:{color:'#5a7a94',font:{size:9},callback:v=>'$'+v.toFixed(0)}}}}
  });
}

// ── WORLD MAP ──────────────────────────────────────────────
const mapEl=document.getElementById('world-map');
const mapSvg=d3.select(mapEl).append('svg').attr('viewBox','0 0 960 500').attr('preserveAspectRatio','xMidYMid meet');
const projection=d3.geoNaturalEarth1().scale(153).translate([480,250]);
const pathGen=d3.geoPath(projection);

const tooltip=document.createElement('div');
tooltip.style.cssText='position:absolute;background:#0b1520;border:1px solid #00ff9c;padding:5px 8px;font-size:9px;color:#c8d8e8;pointer-events:none;display:none;z-index:100;font-family:IBM Plex Mono,monospace;max-width:160px;line-height:1.5';
mapEl.style.position='relative';
mapEl.appendChild(tooltip);

d3.json('https://cdn.jsdelivr.net/npm/world-atlas@2/countries-110m.json').then(world=>{
  const features=topojson.feature(world,world.objects.countries).features;
  mapSvg.selectAll('path').data(features).join('path')
  .attr('d',pathGen)
  .attr('fill',d=>getCountryColor(+d.id))
  .attr('stroke','#1e2d3d')
  .attr('stroke-width',0.5)
  .style('cursor',d=>COUNTRY_DATA[+d.id]?'pointer':'default')
  .on('mousemove',(event,d)=>{
    const cd=COUNTRY_DATA[+d.id];
    if(!cd) return;
    const col=cd.sig==='BULL'?'#00ff9c':cd.sig==='BEAR'?'#ff4d4f':'#f5c842';
    tooltip.style.display='block';
    tooltip.style.left=(event.offsetX+10)+'px';
    tooltip.style.top=(event.offsetY-20)+'px';
    tooltip.innerHTML=`<b style="color:${col}">${cd.name}</b><br>Signal: <span style="color:${col}">${cd.sig}</span> · Score: ${cd.score.toFixed(2)}<br>Sector: ${cd.sector}`;
  })
  .on('mouseleave',()=>{tooltip.style.display='none'})
  .on('click',(event,d)=>{
    const cd=COUNTRY_DATA[+d.id];
    if(!cd) return;
    showCountry(cd);
  });
  
  // Legend
  const lg=mapSvg.append('g').attr('transform','translate(10,460)');
  [{c:'#008855',l:'Bullish'},{c:'#cc2222',l:'Bearish'},{c:'#665500',l:'Neutral'},{c:'#1a2535',l:'Low Data'}].forEach((item,i)=>{
    lg.append('rect').attr('x',i*72).attr('y',0).attr('width',10).attr('height',10).attr('fill',item.c);
    lg.append('text').attr('x',i*72+13).attr('y',9).attr('fill','#5a7a94').attr('font-size','9').attr('font-family','IBM Plex Mono').text(item.l);
  });
}).catch(()=>{
  mapSvg.append('text').attr('x',480).attr('y',250).attr('text-anchor','middle').attr('fill','#5a7a94').attr('font-size','12').text('Map loading...');
});

function showCountry(cd){
  const col=cd.sig==='BULL'?'var(--bull)':cd.sig==='BEAR'?'var(--bear)':'var(--neut)';
  document.getElementById('cp-name').textContent=cd.name;
  document.getElementById('cp-score').innerHTML=`<span style="color:${col}">${cd.score.toFixed(2)}</span>`;
  document.getElementById('cp-signal').innerHTML=`<span style="color:${col}">${cd.sig}</span>`;
  document.getElementById('cp-sector').textContent=cd.sector;
  document.getElementById('cp-trend').innerHTML=`<span style="color:${col}">${cd.trend}</span>`;
  document.getElementById('cp-headlines').innerHTML='<b style="color:#5a7a94;font-size:9px">KEY HEADLINES</b><br>'+cd.headlines.map(h=>`· ${h}`).join('<br>');
  document.getElementById('cp-ai').innerHTML=`<span style="color:#5a7a94;font-size:9px">AI SUMMARY</span><br>${cd.ai}`;
  document.getElementById('country-panel').classList.add('visible');
  document.getElementById('country-panel').scrollIntoView({behavior:'smooth',block:'nearest'});
}
